#include<iostream.h>
#include<stdio.h>
#include<stdlib.h>
#include<malloc.h>
typedef int datatype;
typedef struct node
{
	datatype data;
	struct node *next;
}linklist;
void jay(linklist *a)
{
	if(a->next!=NULL)
	{
		cout<<a->data<<endl;
		jay(a->next);
	}
}
void main()
{
	char ch;
	linklist *head,*s;
	head=NULL;
	cout<<"please enter a charter:"<<endl;
	cin>>ch;
	while(ch!='&')
	{
		s=(linklist *)malloc(sizeof(linklist));//建立新节点
		s->data=ch;//向新节点中添加内容
		s->next=head;//使新节点指向链首
		head=s;//改变头指针
		cout<<"please enter a charter:"<<endl;
		cin>>ch;
	}
	cout<<head->data<<endl;
	jay(s);
}
