#include<iostream.h>
#include<stdlib.h>
#include<stdio.h>
#include<malloc.h>
typedef int datatype;
typedef struct node
{
	datatype data;
	struct node *next;
}linklist;
void main()
{
	linklist *head,*s,*r;
	int i,j,k;
	cout<<"please enter a total number"<<endl;
	cin>>i;
	head=NULL;
	r=NULL;
	for(j=1;j<=i;j++)
	{
		cout<<"please enter a number:"<<endl;
		cin>>k;
		s=(linklist *)malloc(sizeof(linklist));
		s->data=k;
		if(head==NULL) head=s;
		else
			r->next=s;
		r=s;
		cout<<s->data<<endl;
	}
}
